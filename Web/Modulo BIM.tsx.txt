export default function App() {
  // Wizard view tracking: 0 = Landing, 1 = Perfil, 2 = Ciclo Vida, 3 = Procesos, 4 = Recomendación Usos, 5 = Detalle/Ficha, 6 = Resumen
  const [step, setStep] = useState<number>(0);

  // Core Wizard selections
  const [selection, setSelection] = useState<ExtendedSelectionState>({
    actorType: null,
    role: null,
    scale: null,
    phases: {},
    processes: [],
    customFichas: {}
  });

  // Current selected BIM use for specialized editing in Step 5
  const [activeBimUseId, setActiveBimUseId] = useState<string | null>(null);

  // UI state for AI generation
  const [isGenerating, setIsGenerating] = useState<boolean>(false);
  const [generationError, setGenerationError] = useState<string | null>(null);
  const [generationFeedback, setGenerationFeedback] = useState<string>("");

  // Step 3 (Procesos) Sub-navigation state: indexes which phase the user is configuring processes for
  const [activeProcessPhaseIndex, setActiveProcessPhaseIndex] = useState<number>(0);

  // Glossary and inline visual examples states
  const [isGlossaryOpen, setIsGlossaryOpen] = useState<boolean>(false);
  const [expandedVisualUseId, setExpandedVisualUseId] = useState<string | null>(null);

  // Selected phases list sorted by order
  const selectedPhases = useMemo(() => {
    return PHASES.filter(p => selection.phases[p.id] === "trabaja" || selection.phases[p.id] === "critica")
      .sort((a, b) => a.order - b.order);
  }, [selection.phases]);

  // Gantt-style macro-stages groupings matching page 52 from the Planbim Chile Standard
  const ganttGroups = useMemo(() => {
    const groups = [
      {
        id: "planificacion",
        name: "Planificación",
        colorText: "text-[#E61E32]",
        borderColor: "border-[#E61E32]/30",
        bgColor: "bg-[#E61E32]/5",
        headerBg: "bg-[#E61E32]",
        phaseIds: ["planificacion"],
        rangeText: "Usos 1 a 5"
      },
      {
        id: "diseno",
        name: "Diseño",
        colorText: "text-[#b5362e]",
        borderColor: "border-[#b5362e]/30",
        bgColor: "bg-[#b5362e]/5",
        headerBg: "bg-[#b5362e]",
        phaseIds: ["conceptual", "detalle", "documentacion"],
        rangeText: "Usos 6 a 15"
      },
      {
        id: "construccion",
        name: "Construcción",
        colorText: "text-[#e65c50]",
        borderColor: "border-[#e65c50]/30",
        bgColor: "bg-[#e65c50]/5",
        headerBg: "bg-[#e65c50]",
        phaseIds: ["construccion", "puesta"],
        rangeText: "Usos 16 a 20"
      },
      {
        id: "operacion",
        name: "Operación",
        colorText: "text-[#8f2821]",
        borderColor: "border-[#8f2821]/30",
        bgColor: "bg-[#8f2821]/5",
        headerBg: "bg-[#8f2821]",
        phaseIds: ["operacion"],
        rangeText: "Usos 21 a 25"
      }
    ];

    return groups.map(g => {
      const activePhases = selectedPhases.filter(p => g.phaseIds.includes(p.id));
      return {
        ...g,
        activePhases
      };
    });
  }, [selectedPhases]);

  // Handle setting Role
  const handleSelectRole = (role: RoleType) => {
    setSelection(prev => ({ ...prev, role: prev.role === role ? null : role }));
  };

  // Handle setting Scale
  const handleSelectScale = (scale: OperationScale) => {
    setSelection(prev => ({ ...prev, scale: prev.scale === scale ? null : scale }));
  };

  // Toggle stage selection cycle: none -> trabaja -> critica -> none
  const togglePhase = (phaseId: string) => {
    setSelection(prev => {
      const current = prev.phases[phaseId];
      const nextPhases = { ...prev.phases };
      
      if (!current) {
        nextPhases[phaseId] = "trabaja";
      } else if (current === "trabaja") {
        nextPhases[phaseId] = "critica";
      } else {
        delete nextPhases[phaseId];
        // Clean up associated processes
        const phaseProcesses = PROCESSES.filter(p => p.phaseId === phaseId).map(p => p.id);
        prev.processes = prev.processes.filter(pId => !phaseProcesses.includes(pId));
      }
      
      return {
        ...prev,
        phases: nextPhases
      };
    });
  };

  // Directly set a phase as Critical
  const togglePhaseCritical = (e: React.MouseEvent, phaseId: string) => {
    e.stopPropagation();
    setSelection(prev => {
      const current = prev.phases[phaseId];
      const nextPhases = { ...prev.phases };
      
      if (current === "critica") {
        nextPhases[phaseId] = "trabaja";
      } else {
        nextPhases[phaseId] = "critica";
      }
      
      return {
        ...prev,
        phases: nextPhases
      };
    });
  };

  // Toggle active process selection
  const toggleProcess = (processId: string) => {
    setSelection(prev => {
      const isSelected = prev.processes.includes(processId);
      const nextProcesses = isSelected 
        ? prev.processes.filter(id => id !== processId)
        : [...prev.processes, processId];
      return {
        ...prev,
        processes: nextProcesses
      };
    });
  };

  // Recommendation engine: calculates scores for each BIM Use
  const scoredBimUses = useMemo(() => {
    return BIM_USES.map(use => {
      let score = 0;
      let reasons: string[] = [];

      // 1. Role match
      if (selection.role && use.roles.includes(selection.role)) {
        score += 6;
        reasons.push(`Alineado con tu rol de ${selection.role}`);
      }

      // 2. Phase match
      const matchingPhases = use.phases.filter(pId => selection.phases[pId]);
      matchingPhases.forEach(pId => {
        const pond = selection.phases[pId];
        const phaseName = PHASES.find(p => p.id === pId)?.name || "";
        if (pond === "critica") {
          score += 8;
          reasons.push(`Fase crítica: ${phaseName} (+8 pts)`);
        } else if (pond === "trabaja") {
          score += 4;
          reasons.push(`Trabajas en: ${phaseName} (+4 pts)`);
        }
      });

      // 3. Process matches
      const useProcesses = use.processes; // associated key processes
      const matchingProcesses = selection.processes.filter(procId => useProcesses.includes(procId));
      if (matchingProcesses.length > 0) {
        const pts = matchingProcesses.length * 6;
        score += pts;
        reasons.push(`${matchingProcesses.length} procesos clave vinculados (+${pts} pts)`);
      }

      // 4. Actor Type customized matching
      if (selection.actorType === "Mandante") {
        if (use.id === "estimacion_5d") {
          score += 10;
          reasons.push("Prioridad Mandante: Control presupuestario (+10 pts)");
        } else if (use.id === "mantenimiento_activos") {
          score += 10;
          reasons.push("Prioridad Mandante: Gestión de Activos y FM (+10 pts)");
        } else if (use.id === "as_built_registro") {
          score += 8;
          reasons.push("Prioridad Mandante: Recepción As-Built (+8 pts)");
        } else if (use.id === "coordinacion_3d") {
          score += 5;
          reasons.push("Prioridad Mandante: Aseguramiento de Calidad (+5 pts)");
        }
      } else if (selection.actorType === "Consultor") {
        if (use.id === "coordinacion_3d") {
          score += 10;
          reasons.push("Prioridad Consultor: Coordinación espacial multidisciplinar (+10 pts)");
        } else if (use.id === "planificacion_4d") {
          score += 8;
          reasons.push("Prioridad Consultor: Planificación de plazos constructivos (+8 pts)");
        } else if (use.id === "levantamiento") {
          score += 6;
          reasons.push("Prioridad Consultor: Levantamiento de condiciones existentes (+6 pts)");
        }
      }

      return {
        use,
        score,
        reasons
      };
    }).sort((a, b) => b.score - a.score);
  }, [selection.actorType, selection.role, selection.phases, selection.processes]);

  // Divide into priority clusters
  const recommendations = useMemo(() => {
    const highest = scoredBimUses.filter(item => item.score >= 12);
    const suggested = scoredBimUses.filter(item => item.score >= 5 && item.score < 12);
    const rest = scoredBimUses.filter(item => item.score < 5);
    return { highest, suggested, rest };
  }, [scoredBimUses]);

  // Automatically opt-in recommended BIM uses in secondary list or report state
  const [selectedBimUseIds, setSelectedBimUseIds] = useState<Record<string, boolean>>({});

  // When step 4 is entered, pre-initialize selection based on processes selected in step 3
  React.useEffect(() => {
    if (step === 4) {
      setSelectedBimUseIds(prev => {
        const updated = { ...prev };
        BIM_USES.forEach(use => {
          if (selection.processes.includes(use.id)) {
            updated[use.id] = true;
          }
        });
        return updated;
      });
    }
  }, [step, selection.processes]);

  const toggleBimUseSelection = (useId: string) => {
    setSelectedBimUseIds(prev => {
      const isCurrentlySelected = !!prev[useId];
      const nextSelection = !isCurrentlySelected;
      
      // If we are deselecting, let's also remove its custom ficha to be coherent
      if (!nextSelection) {
        setSelection(selPrev => {
          const nextCustom = { ...selPrev.customFichas };
          delete nextCustom[useId];
          return {
            ...selPrev,
            customFichas: nextCustom
          };
        });
      }
      return {
        ...prev,
        [useId]: nextSelection
      };
    });
  };

  // Get active BIM use object
  const activeBimUse = useMemo(() => {
    return BIM_USES.find(u => u.id === activeBimUseId) || null;
  }, [activeBimUseId]);

  // Trigger Ficha derivation via Gemini API
  const generateFichaWithAI = async (useInfo: BimUseInfo) => {
    setIsGenerating(true);
    setGenerationError(null);
    setGenerationFeedback("Calculando proyecciones de inversión y estandarización técnica...");

    const phaseLabels = useInfo.phases.map(pId => PHASES.find(p => p.id === pId)?.name || pId).join(", ");
    const selectedProcsNames = PROCESSES.filter(p => selection.processes.includes(p.id) && useInfo.processes.includes(p.id))
      .map(p => p.name);

    setTimeout(() => setGenerationFeedback("Sincronizando flujos según estándares de intercambio ISO 19650..."), 800);
    setTimeout(() => setGenerationFeedback("Evaluando indicadores CAPEX/OPEX y esfuerzos de adopción..."), 1600);
    setTimeout(() => setGenerationFeedback("Estructurando criterios de éxito económico del proyecto..."), 2400);

    try {
      const res = await fetch("/api/generate-ficha", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          bimUse: useInfo.name,
          rol: selection.role,
          scale: selection.scale,
          phase: phaseLabels,
          processes: selectedProcsNames
        })
      });

      if (!res.ok) {
        throw new Error("Respuesta de servidor no satisfactoria.");
      }

      const data = await res.json();
      
      setSelection(prev => {
        const nextCustom = { ...prev.customFichas };
        nextCustom[useInfo.id] = {
          objetivo: data.objetivo || "",
          responsables: data.responsables || "",
          lodLoin: data.lodLoin || "",
          entregables: data.entregables || "",
          criteriosExito: data.criteriosExito || "",
          costoImplementacion: data.costoImplementacion || "",
          roiEstimado: data.roiEstimado || "",
          esfuerzoAdopcion: data.esfuerzoAdopcion || ""
        };
        return {
          ...prev,
          customFichas: nextCustom
        };
      });

      setGenerationFeedback("¡Evaluación técnica y económica generada con éxito!");
      setTimeout(() => setIsGenerating(false), 500);

    } catch (err: any) {
      console.error(err);
      setGenerationError("Ocurrió un error al contactar al motor de IA. Se ha cargado la estructura de evaluación base.");
      loadDefaultFicha(useInfo);
      setIsGenerating(false);
    }
  };

  const loadDefaultFicha = (useInfo: BimUseInfo) => {
    const processesJoined = PROCESSES.filter(p => selection.processes.includes(p.id))
      .map(p => p.name)
      .slice(0, 3)
      .join(", ") || "Optimización de flujos informativos";

    const defaultFicha: ExtendedBimFicha = {
      objetivo: `Implementar el Uso BIM "${useInfo.name}" para resolver desvíos y optimizar el flujo informátivo en las fases de ${useInfo.phases.map(pid => PHASES.find(p => p.id === pid)?.name).join(", ")}. Configurado para el rol de ${selection.role || "Especialista AEC"}.`,
      responsables: `BIM Manager, Coordinador de Especialidades, y el rol estratégico de ${selection.role || "Representante Técnico"}.`,
      lodLoin: selection.scale?.includes("Grande") 
        ? "LOD 350 / LOIN Alto - Parámetros geométricos exactos, enlaces de adquisiciones y atributos de operación continua."
        : selection.scale?.includes("Mediana")
        ? "LOD 300 / LOIN Medio - Dimensiones precisas, ensambles estructurales e información específica de fabricantes."
        : "LOD 200 / LOIN Básico - Geometría aproximada con fines de asignación e interferencia espacial preliminar.",
      entregables: "Modelo federado estructurado en estándar abierto (.IFC) o formato nativo unificado, informe consolidado de desvíos, matrices de extracción de cantidades (cubicaciones).",
      criteriosExito: `1. Reducción estimada de un 15% en procesos redundantes asociados a: ${processesJoined}.
2. Eliminación de asimetrías de información críticas dentro del Entorno de Datos Comunes (CDE).`,
      costoImplementacion: selection.scale?.includes("Grande") 
        ? "Mínimo de Infraestructura de SOFTWARE: Autodesk AEC Collection (Revit, Navisworks), Solibri. HARDWARE: Estaciones de trabajo Xeon/Core i9, GPU RTX (8GB+ VRAM), 64GB RAM. Inversión Alta (CAPEX/OPEX): Requiere infraestructura robusta de servidores en la nube (CDE corporativo) y licenciamiento avanzado integrado."
        : selection.scale?.includes("Mediana")
        ? "Mínimo de Infraestructura de SOFTWARE: Autodesk Revit, Navisworks Simulate, Archicad. HARDWARE: PC Core i7/Ryzen 7, GPU dedicada (6GB), 32GB RAM, disco SSD. Inversión Moderada: Licenciamiento standalone estándar, uso de plataformas colaborativas en la nube."
        : "Mínimo de Infraestructura de SOFTWARE: FreeCAD, SketchUp Pro, Autodesk Revit LT, visores gratuitos. HARDWARE: PC Core i5/Ryzen 5, GPU dedicada básica (4GB VRAM), 16GB RAM. Inversión Baja: Flujos de colaboración directa simplificados.",
      roiEstimado: selection.actorType === "Mandante"
        ? "Retorno Alto (Foco OPEX): Reducción drástica de órdenes de cambio imprevistas durante la construcción y control presupuestario optimizado del activo. Amortización estimada dentro del primer año de explotación de la infraestructura."
        : "Retorno Técnico (Foco CAPEX): Mitigación de horas de re-diseño por colisiones multidisciplinares no detectadas. Incremento en la tasa de adjudicación de licitaciones públicas al cumplir con exigencias Planbim.",
      esfuerzoAdopcion: selection.scale?.includes("Grande")
        ? "Personal (RRHH) requerido: BIM Manager Senior especialista, 1 Coordinador BIM dedicado por proyecto y 2 a 3 Modeladores BIM certificados con experiencia en ISO 19650. Curva de aprendizaje e implementación de alta complejidad (6 a 12 meses)."
        : selection.scale?.includes("Mediana")
        ? "Personal (RRHH) requerido: 1 Coordinador BIM a tiempo parcial (híbrido) y 1 a 2 Modeladores BIM capacitados. Curva de aprendizaje moderada con plantillas y manuales estándar (2 a 3 meses)."
        : "Personal (RRHH) requerido: 1 Diseñador/Modelador con capacitación básica en modelamiento paramétrico y un consultor externo para auditorías periódicas. Curva de adopción simple."
    };

    setSelection(prev => {
      const nextCustom = { ...prev.customFichas };
      nextCustom[useInfo.id] = defaultFicha;
      return {
        ...prev,
        customFichas: nextCustom
      };
    });
  };

  const handleUpdateFichaField = (useId: string, field: keyof ExtendedBimFicha, value: string) => {
    setSelection(prev => {
      const current = prev.customFichas[useId] || {
        objetivo: "", responsables: "", lodLoin: "", entregables: "", criteriosExito: "", costoImplementacion: "", roiEstimado: "", esfuerzoAdopcion: ""
      };
      
      const nextCustom = { ...prev.customFichas };
      nextCustom[useId] = {
        ...current,
        [field]: value
      };

      return {
        ...prev,
        customFichas: nextCustom
      };
    });
  };

  const ensureFichaExists = (useInfo: BimUseInfo) => {
    if (!selection.customFichas[useInfo.id]) {
      loadDefaultFicha(useInfo);
    }
  };

  const openBimUseDetail = (useInfo: BimUseInfo) => {
    setActiveBimUseId(useInfo.id);
    ensureFichaExists(useInfo);
    setSelectedBimUseIds(prev => ({
      ...prev,
      [useInfo.id]: true
    }));
    setStep(5);
  };

  const handleReset = () => {
    setSelection({
      actorType: null,
      role: null,
      scale: null,
      phases: {},
      processes: [],
      customFichas: {}
    });
    setSelectedBimUseIds({});
    setActiveBimUseId(null);
    setActiveProcessPhaseIndex(0);
    setStep(0);
  };

  const canGoNext = () => {
    if (step === 1) return selection.actorType !== null && selection.role !== null && selection.scale !== null;
    if (step === 2) return selectedPhases.length > 0;
    if (step === 3) return selection.processes.length > 0;
    return true;
  };

  const handleNext = () => {
    if (step === 1) {
      setStep(2);
    } else if (step === 2) {
      setActiveProcessPhaseIndex(0);
      setStep(3);
    } else if (step === 3) {
      setStep(4);
    } else if (step === 4) {
      setStep(6);
    } else if (step === 5) {
      setStep(4);
    }
  };

  const handleBack = () => {
    if (step === 1) setStep(0);
    else if (step === 2) setStep(1);
    else if (step === 3) setStep(2);
    else if (step === 4) setStep(3);
    else if (step === 5) setStep(4);
    else if (step === 6) setStep(4);
  };

  const progressPercent = useMemo(() => {
    if (step === 0) return 5;
    return Math.min(100, Math.round((step / 6) * 100));
  }, [step]);

  const finalSelectedUses = useMemo(() => {
    return BIM_USES.filter(u => selectedBimUseIds[u.id]);
  }, [selectedBimUseIds]);

  return (
    <div className="min-h-screen bg-white text-[#222222] flex flex-col font-sans transition-colors duration-200 relative select-none">
      
      {/* Layout Estético Subliminal */}
      <div className="absolute inset-0 pointer-events-none opacity-[0.03] mix-blend-multiply bg-[url('https://www.transparenttextures.com/patterns/natural-paper.png')] z-0"></div>
      
      <div className="absolute bottom-24 right-12 pointer-events-none select-none opacity-[0.02] z-0 no-print">
        <span className="text-[260px] font-serif leading-none">墨</span>
      </div>

      {/* Cabecera Global de Progreso */}
      {step > 0 && (
        <header className="pt-4 pb-2 px-[1cm] flex flex-col md:flex-row items-start md:items-center justify-between border-b border-neutral-250 bg-white z-40 no-print sticky top-0 shadow-sm">
          <div className="flex flex-col mb-2 md:mb-0">
            <div className="flex items-center gap-2">
              <span className="text-[10px] uppercase tracking-[0.2em] font-bold text-neutral-500">
                Paso 0{step} de 06
              </span>
              <span className="text-xs text-[#E61E32]">|</span>
              <span className="text-xs text-neutral-700 font-medium">Asistente de Inversión y Madurez</span>
              <span className="text-xs text-neutral-300">|</span>
              <button
                onClick={() => setIsGlossaryOpen(true)}
                className="text-xs text-neutral-800 hover:text-[#E61E32] font-bold flex items-center gap-1 cursor-pointer transition-all"
                title="Abrir Glosario Técnico"
              >
                <BookOpen size={13} />
                <span>Glosario BIM Normativo</span>
              </button>
            </div>
            <h1 className="text-lg md:text-xl font-light italic tracking-tight font-serif text-neutral-900 mt-0.5">
              {step === 1 && "Definición del Perfil del Profesional"}
              {step === 2 && "Mapeo del Ciclo de Vida de los Proyectos"}
              {step === 3 && "Identificación Operativa de Procesos Clave"}
              {step === 4 && "Taxonomía y Factibilidad de Usos BIM"}
              {step === 5 && `Análisis Técnico-Económico: ${activeBimUse?.name}`}
              {step === 6 && "Hoja de Ruta e Informe de Viabilidad Consolidado"}
            </h1>

              </button>

            {step === 6 && (
              <button onClick={() => setStep(4)} className="flex items-center gap-1.5 px-5 py-3 border border-neutral-900 text-neutral-900 hover:bg-neutral-950 hover:text-white rounded-[4px] text-[11px] font-bold uppercase cursor-pointer">
                Ajustar Parámetros
              </button>
            )}
          </div>
        </footer>
      )}

      {step === 0 && (
        <footer className="w-full no-print py-4 px-[1cm] border-t border-[#f2f1eb]/60 bg-gray-50/40 text-center mt-auto z-10">
          <div className="max-w-4xl mx-auto flex flex-col md:flex-row items-center justify-between gap-4">
            <p className="text-[10px] font-mono tracking-wide text-gray-400 uppercase">Planbim CORFO & Penn State Ecosistema Chileno</p>
            <p className="text-[11px] text-gray-500 font-light text-right leading-relaxed max-w-2xl">
              Este asistente parametriza la Solicitud de Información (SDI) y el Plan de Ejecución BIM (PEB) cruzando la complejidad del activo con el esfuerzo de cambio organizacional y financiero exigido.
            </p>
          </div>
        </footer>
      )}

      <button onClick={() => setIsGlossaryOpen(true)} className="fixed bottom-6 right-6 z-40 bg-[#E61E32] text-white p-3.5 shadow-xl hover:bg-[#b8332b] rounded-full border border-white/15 flex items-center justify-center cursor-pointer group no-print">
        <BookOpen size={20} />
        <span className="max-w-0 overflow-hidden group-hover:max-w-24 group-hover:ml-2 transition-all duration-300 text-[10px] font-mono uppercase tracking-widest whitespace-nowrap">Glosario</span>
      </button>

      <Glossary isOpen={isGlossaryOpen} onClose={() => setIsGlossaryOpen(false)} />
    </div>
  );
}
